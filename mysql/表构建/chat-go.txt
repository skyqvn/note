CREATE TABLE IF NOT EXISTS `ChatGroup`(
`id` INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
`name` char(25),
`password` char(20),
`introduce` char(200)
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE IF NOT EXISTS `Report`(
`chatgroup` int unsigned,
`username` char(25),
`value` text,
`date` datetime default CURRENT_TIMESTAMP
on update CURRENT_TIMESTAMP,
foreign key(chatgroup) references ChatGroup(id) on delete cascade
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE IF NOT EXISTS `User`(
`id` INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
`name` char(25) not null default '匿名用户',
`password` char(20) not null default '000000',
`introduce` char(200)
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE IF NOT EXISTS `Member`(
`owner` int unsigned,
`group` int unsigned,
foreign key(`group`) references ChatGroup(id)
on delete cascade,
foreign key(owner) references User(`id`)
on delete cascade
)ENGINE=InnoDB DEFAULT CHARSET=utf8;
